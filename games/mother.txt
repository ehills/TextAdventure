gameName = "*****Mother(Example Game)*****";
credits = "By randoms";

Player mother{
       name = "Mother";
       description: "A concerned Mother";
       carryLimit = 4;
}

ItemDefaults {
    Verb "examine" {
        print inputItem.description();
    }
    Verb "pickup" {
        if (mother canCarry) {
            inputItem setLocation mother ;
        } else {
            print "You are carrying too much already";
        }
    }

    Verb "drop" {
        if (mother hasItem inputItem ) {
            inputItem setLocation currentLocation;
        } else {
            print "I don't have ";
            print inputItem;
            print ".";
        }
    }
}

LocationDefaults {
	Verb "i or inventory or invent" {
		describeInventory;
	}
    Verb "look" {
			describe;
			list;	
    }
    Verb "quit or q" {
    	print "Bye bye thanks for playing!!!";
    	gameOver;
    }   
    // Directions
    Verb "north or n or nth" {
    	if(mother inLocation hasNorth) {
    		mother setLocation toNorth;
    		describe;
			list;
    	} else {
    		print "You cannot go north.";
    	}
    }
    Verb "south or s or sth" {
    	if(mother inLocation hasSouth) {
    		mother setLocation toSouth;
    		describe;
			list;
    	} else {
    		print "You cannot go south.";
    	}
    }
    Verb "east or e" {
    	if(mother inLocation hasEast) {
    		mother setLocation toEast;
    		describe;
			list;
    	} else {
    		print "You cannot go east.";
    	}

    Verb "west or w" {
    	if(mother inLocation hasWest) {
    		mother setLocation toWest;
    		describe;
			list;
    	} else {
    		print "You cannot go west.";
    	}
    }
}

                print "The toilet is empty";
            }
